001package io.github.flowersinthesand.wes.atmosphere;
002
003import io.github.flowersinthesand.wes.Data;
004import io.github.flowersinthesand.wes.http.AbstractServerHttpExchange;
005import io.github.flowersinthesand.wes.http.ServerHttpExchange;
006import io.github.flowersinthesand.wes.http.StatusCode;
007
008import java.io.IOException;
009import java.io.PrintWriter;
010import java.util.ArrayList;
011import java.util.Collections;
012import java.util.Enumeration;
013import java.util.LinkedHashSet;
014import java.util.List;
015import java.util.Set;
016
017import javax.servlet.ReadListener;
018import javax.servlet.ServletInputStream;
019
020import org.atmosphere.cpr.AtmosphereResource;
021import org.atmosphere.cpr.AtmosphereResourceEvent;
022import org.atmosphere.cpr.AtmosphereResourceEventListenerAdapter;
023
024public class AtmosphereServerHttpExchange extends AbstractServerHttpExchange {
025
026    private final AtmosphereResource resource;
027
028    public AtmosphereServerHttpExchange(AtmosphereResource resource) {
029        this.resource = resource.suspend();
030        resource.addEventListener(new AtmosphereResourceEventListenerAdapter() {
031            @Override
032            public void onResume(AtmosphereResourceEvent event) {
033                closeActions.fire();
034            }
035
036            @Override
037            public void onDisconnect(AtmosphereResourceEvent event) {
038                closeActions.fire();
039            }
040
041            @Override
042            public void onClose(AtmosphereResourceEvent event) {
043                closeActions.fire();
044            }
045        });
046    }
047
048    @Override
049    public String uri() {
050        return resource.getRequest().getRequestURI();
051    }
052
053    @Override
054    public String method() {
055        return resource.getRequest().getMethod();
056    }
057
058    @Override
059    public Set<String> requestHeaderNames() {
060        Set<String> headerNames = new LinkedHashSet<>();
061        Enumeration<String> enumeration = resource.getRequest().getHeaderNames();
062        while (enumeration.hasMoreElements()) {
063            headerNames.add(enumeration.nextElement());
064        }
065        return headerNames;
066    }
067
068    @SuppressWarnings("unchecked")
069    @Override
070    public List<String> requestHeaders(String name) {
071        return Collections.list(resource.getRequest().getHeaders(name));
072    }
073    
074    @Override
075    protected void readBody() {
076        try {
077            final ServletInputStream input = resource.getRequest().getInputStream();
078            // Since Servlet 3.1
079            input.setReadListener(new ReadListener() {
080                List<String> chunks = new ArrayList<>();
081                @Override
082                public void onDataAvailable() throws IOException {
083                    int bytesRead = -1;
084                    byte buffer[] = new byte[4096];
085                    while (input.isReady() && (bytesRead = input.read(buffer)) != -1) {
086                        String data = new String(buffer, 0, bytesRead);
087                        chunks.add(data);
088                    }
089                }
090
091                @Override
092                public void onAllDataRead() throws IOException {
093                    StringBuilder body = new StringBuilder();
094                    for (String chunk : chunks) {
095                        body.append(chunk);
096                    }
097                    bodyActions.fire(new Data(body.toString()));
098                }
099
100                @Override
101                public void onError(Throwable t) {}
102            });
103        } catch (IOException e) {
104            throw new RuntimeException(e);
105        }
106    }
107
108    @Override
109    public ServerHttpExchange setResponseHeader(String name, String value) {
110        resource.getResponse().setHeader(name, value);
111        return this;
112    }
113
114    @Override
115    public ServerHttpExchange setStatus(StatusCode status) {
116        resource.getResponse().setStatus(status.code(), status.reason());
117        return this;
118    }
119
120    @Override
121    protected void doWrite(String data) {
122        try {
123            PrintWriter writer = resource.getResponse().getWriter();
124            writer.print(data);
125            writer.flush();
126        } catch (IOException e) {
127            throw new RuntimeException(e);
128        }
129    }
130
131    @Override
132    protected void doClose() {
133        resource.resume();
134        try {
135            resource.close();
136        } catch (IOException e) {}
137    }
138
139    @Override
140    public <T> T unwrap(Class<T> clazz) {
141        return AtmosphereResource.class.isAssignableFrom(clazz) ? clazz.cast(resource) : null;
142    }
143
144}
